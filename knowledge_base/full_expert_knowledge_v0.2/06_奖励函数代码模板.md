# 06_奖励函数代码模板

## 1. 核心要求

LLM 生成奖励代码时必须满足：

```text
1. 不允许只返回一个总 reward。
2. 必须记录 reward_terms。
3. 每个奖励项必须对应一个 skeleton_id。
4. 每个奖励项必须写注释说明作用。
5. 不允许使用环境中不存在的变量。
6. 不允许使用未来信息。
7. 不允许使用训练时不可获得的信息。
8. 必须考虑 NaN/Inf 和尺度问题。
```

## 2. 通用 RewardFunction 模板

```python
import numpy as np

class RewardFunction:
    def __init__(self, config=None):
        config = config or {}
        self.w_success = config.get("w_success", 100.0)
        self.w_failure = config.get("w_failure", 100.0)
        self.w_progress = config.get("w_progress", 1.0)
        self.w_alive = config.get("w_alive", 0.1)
        self.w_energy = config.get("w_energy", 0.001)
        self.w_smooth = config.get("w_smooth", 0.001)
        self.prev_distance = None
        self.prev_action = None

    def reset(self):
        self.prev_distance = None
        self.prev_action = None

    def safe_clip(self, x, low=-10.0, high=10.0):
        x = float(x)
        if np.isnan(x) or np.isinf(x):
            return 0.0
        return float(np.clip(x, low, high))

    def compute_reward(self, obs, action, next_obs, done, info):
        reward_terms = {}
        reward_meta = {}

        success = bool(info.get("success", False))
        reward_terms["success"] = self.w_success if success else 0.0
        reward_meta["success"] = {"skeleton_id": "terminal_success_reward", "role": "task_objective"}

        failure = bool(info.get("failure", False))
        reward_terms["failure"] = -self.w_failure if failure else 0.0
        reward_meta["failure"] = {"skeleton_id": "terminal_failure_penalty", "role": "safety_constraint"}

        curr_distance = info.get("distance_to_goal", None)
        if curr_distance is not None and self.prev_distance is not None:
            progress = self.safe_clip(self.prev_distance - curr_distance, -1.0, 1.0)
            reward_terms["progress"] = self.w_progress * progress
        else:
            reward_terms["progress"] = 0.0
        if curr_distance is not None:
            self.prev_distance = float(curr_distance)
        reward_meta["progress"] = {"skeleton_id": "progress_delta_reward", "role": "process_guidance"}

        reward_terms["alive"] = self.w_alive if (not done and not failure) else 0.0
        reward_meta["alive"] = {"skeleton_id": "alive_bonus", "role": "survival_objective"}

        action = np.asarray(action, dtype=np.float32)
        energy_cost = self.safe_clip(np.sum(np.square(action)), 0.0, 100.0)
        reward_terms["energy"] = -self.w_energy * energy_cost
        reward_meta["energy"] = {"skeleton_id": "energy_penalty", "role": "behavior_regularization"}

        if self.prev_action is not None:
            smooth_cost = self.safe_clip(np.sum(np.square(action - self.prev_action)), 0.0, 100.0)
            reward_terms["smoothness"] = -self.w_smooth * smooth_cost
        else:
            reward_terms["smoothness"] = 0.0
        self.prev_action = action.copy()
        reward_meta["smoothness"] = {"skeleton_id": "action_smoothness_penalty", "role": "behavior_regularization"}

        reward = float(sum(reward_terms.values()))
        if np.isnan(reward) or np.isinf(reward):
            reward = 0.0

        info["reward_terms"] = reward_terms
        info["reward_meta"] = reward_meta
        info["reward_total"] = reward
        return reward
```

## 3. Gymnasium step 接入模板

```python
def step(self, action):
    obs_before = self._get_obs()
    self._apply_action(action)
    self._simulate()
    obs_after = self._get_obs()

    success = self._is_success()
    failure = self._is_failure()
    terminated = success or failure
    truncated = self.step_count >= self.max_steps

    info = {
        "success": success,
        "failure": failure,
        "distance_to_goal": self._distance_to_goal(),
        "termination_reason": self._get_termination_reason(),
    }

    reward = self.reward_fn.compute_reward(
        obs=obs_before,
        action=action,
        next_obs=obs_after,
        done=terminated or truncated,
        info=info,
    )

    return obs_after, reward, terminated, truncated, info
```

## 5. 代码审查清单

```yaml
reward_code_review:
  是否记录 reward_terms: true/false
  是否记录 reward_meta: true/false
  是否使用未定义变量: true/false
  是否使用未来信息: true/false
  是否存在 NaN/Inf 风险: true/false
  是否有 clip/scale: true/false
  是否每项有 skeleton_id: true/false
  是否 success/failure 与环境终止条件一致: true/false
  是否可能原地不动刷分: low/medium/high
  是否可能猛冲失败: low/medium/high
```
